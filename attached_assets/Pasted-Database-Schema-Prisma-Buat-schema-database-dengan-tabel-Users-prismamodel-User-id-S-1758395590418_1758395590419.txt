Database Schema (Prisma)
Buat schema database dengan tabel:
Users
prismamodel User {
  id          String   @id @default(cuid())
  email       String   @unique
  nim         String   @unique
  name        String
  prodi       String
  gender      String
  phone       String?
  role        Role     @default(VOTER)
  hasVoted    Boolean  @default(false)
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
  
  votingSessions VotingSession[]
  votes          Vote[]
  adminLogs      AdminLog[]
}

enum Role {
  VOTER
  ADMIN
  SUPER_ADMIN
}
Candidates
prismamodel Candidate {
  id          String @id @default(cuid())
  name        String
  nim         String @unique
  prodi       String
  visi        String @db.Text
  misi        String @db.Text
  photo       String?
  isActive    Boolean @default(true)
  createdAt   DateTime @default(now())
  
  votes       Vote[]
}
VotingSession
prismamodel VotingSession {
  id          String   @id @default(cuid())
  userId      String
  qrCode      String   @unique
  redeemCode  String   @unique
  isValidated Boolean  @default(false)
  isUsed      Boolean  @default(false)
  validatedBy String?
  expiresAt   DateTime
  createdAt   DateTime @default(now())
  
  user        User     @relation(fields: [userId], references: [id])
  validator   User?    @relation(fields: [validatedBy], references: [id])
}
Vote
prismamodel Vote {
  id          String   @id @default(cuid())
  userId      String
  candidateId String
  createdAt   DateTime @default(now())
  
  user        User      @relation(fields: [userId], references: [id])
  candidate   Candidate @relation(fields: [candidateId], references: [id])
}
AdminLog
prismamodel AdminLog {
  id        String   @id @default(cuid())
  adminId   String
  action    String
  target    String?
  details   Json?
  ipAddress String?
  createdAt DateTime @default(now())
  
  admin     User     @relation(fields: [adminId], references: [id])
}
Settings
prismamodel Settings {
  id           String   @id @default(cuid())
  key          String   @unique
  value        String
  description  String?
  updatedAt    DateTime @updatedAt
}
